class A
{
    public static void main(String args[])
    {
        int arr[] = {7,5,6,1,4,2};
        
        int n=arr.length+1;
        int sum = n*(n+1)/2;
        int realSum=0;
        for(int i=0;i<arr.length;i++)
        {
            realSum+= arr[i];
        }
        
        System.out.println("Missing number is : "+(sum-realSum));
    }
}
